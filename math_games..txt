let square = document.getElementById('square');
let triangle = document.getElementById('triangle');
let squareTarget = document.getElementById('squareTarget');
let triangleTarget = document.getElementById('triangleTarget');
let messageBuilder = document.getElementById('messageBuilder');

let placed = {square: false, triangle: false};

function allowDrop(ev) {
  ev.preventDefault();
}

function drag(ev) {
  ev.dataTransfer.setData("text", ev.target.id);
}

function drop(ev, targetShape) {
  ev.preventDefault();
  let data = ev.dataTransfer.getData("text");
  if(data === targetShape) {
    let shape = document.getElementById(data);
    let rect = ev.target.getBoundingClientRect();
    shape.style.position = "absolute";
    shape.style.left = rect.left + "px";
    shape.style.top = rect.top + "px";
    ev.target.appendChild(shape);
    placed[targetShape] = true;
    checkWin();
  } else {
    messageBuilder.textContent = "–¶—è —Ñ—ñ–≥—É—Ä–∞ –Ω–µ –ø—ñ–¥—Ö–æ–¥–∏—Ç—å! –°–ø—Ä–æ–±—É–π —ñ–Ω—à—É.";
  }
}

function checkWin() {
  if(placed.square && placed.triangle) {
    messageBuilder.textContent = "–í—ñ—Ç–∞—î–º–æ! –ë—É–¥–∏–Ω–æ–∫ –≥–æ—Ç–æ–≤–∏–π! üê±üè†";
  }
}

square.addEventListener('dragstart', drag);
triangle.addEventListener('dragstart', drag);

squareTarget.addEventListener('dragover', allowDrop);
triangleTarget.addEventListener('dragover', allowDrop);

squareTarget.addEventListener('drop', (ev) => drop(ev, 'square'));
triangleTarget.addEventListener('drop', (ev) => drop(ev, 'triangle'));

function resetBuilder() {
  placed = {square: false, triangle: false};
  messageBuilder.textContent = "";
  document.body.appendChild(square);
  document.body.appendChild(triangle);
  square.style.position = "static";
  triangle.style.position = "static";
}
